简单实现 模板字符串 的功能

<!-- more --> 

```js
const str = 'i am {{name}},age {{age}},job {{job}} ';
const data = {
  name:'xbd',
  age:18,
  job:'CTO'
}

const func = (str, data) => {
  return str.replace(/\{\{(\w+)\}\}/g, (match, key) => {
    return data[key];
  })
}
console.log(func(str, data));

// \w 数字，字母 ，下划线
// + 1到多个
```
