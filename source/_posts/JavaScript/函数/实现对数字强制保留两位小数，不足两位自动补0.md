```js
function returnFloat(value) {
    let res = Math.round(parseFloat(value) * 100) / 100;
    let s = res.toString().split(".");
    if (s.length == 1) {
        return res.toString() + ".00";
    }
    if (s.length > 1) {
        if (s[1].length < 2) {
            return res.toString() + "0";
        }
        return res.toString()
    }
}
```
